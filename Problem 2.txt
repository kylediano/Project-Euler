nums = [1, 2]
total = 0
i = 0
new_num = 0
does_exceed = False
while not does_exceed:  # fibonacci sequence
    new_num = nums[-1] + nums[-2]
    if new_num > 4000000:
        does_exceed = True
    else:
        nums.append(new_num)
for index in nums:
    if index % 2 == 0:
        total += index
print(nums)
print(total)
